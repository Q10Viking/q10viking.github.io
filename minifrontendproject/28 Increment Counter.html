# 如何并行的？

程序不应该按顺序执行的吗？利用定时器，可以达到并发的效果。主程序主要是设置定时器。接下的程序则由定时器自己运行

const counters = document.querySelectorAll(".counter")!

counters.forEach(counter => {
  counter.innerHTML = "0"
  const target = +counter.getAttribute("data-target")!
  const step = target / 200
  let val = 0

  // 这段函数设置在这里的意义是方便访问外边的变量
  const updateNumber = ()=>{
    val += step
    if(val<target){
      counter.innerHTML = val+""
      // 递归调用 setTimeout这里并不会第一次阻塞下一次的循环，程序设置完定时器后继续执行了
      setTimeout(updateNumber,1)
    }else{
      counter.innerHTML = target+""
    }
  }
  
  updateNumber()
})

# Vue3中使用ref属性

Template Refopen in new window

注意：名字要相同

官网的建议

<script setup lang="ts">
import { ref, onMounted } from 'vue'

const el = ref<HTMLInputElement | null>(null)

onMounted(() => {
  el.value?.focus()
})
</script>

<template>
  <input ref="el" />
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
// 定义
const tiktok = ref<HTMLParamElement | null>(null)

onMounted(() => {
  // 使用获取属性
  tiktok.value?.getAttribute("data-target")
})
</script>

<template>
  <div class="item">
        <font-awesome-icon icon="fa-brands fa-tiktok" class="icon"/>
        <p class="counter" 
            data-target="23000"
            ref="tiktok"></p>   <!-- 名字要相同 -->
        <span>Tiktok Fans</span>
      </div>
</template>
