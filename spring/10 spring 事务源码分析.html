# 提交时强制回滚

@Transactional
public void test(){
    jdbcTemplate.execute
    try{
        //比如远程调用rpc call
        throw new NullPointerException();
    }catch(Exception e){
        TransactionAspectSupport.currentTransactionStatus().setRollbackOnly
        return new Result()
    }
}
