一段java和go多线程对比的代码，它们都是用四个线程（或协程）来计算一个数组的和：

// Java
public class SumArray {
  // 定义一个数组
  static int[] array = {1, 2, 3, 4, 5, 6, 7, 8};
  // 定义一个结果变量
  static int sum = 0;
  // 定义一个锁对象
  static Object lock = new Object();

  public static void main(String[] args) throws InterruptedException {
    // 创建四个线程
    Thread t1 = new Thread(new SumTask(0, 1));
    Thread t2 = new Thread(new SumTask(2, 3));
    Thread t3 = new Thread(new SumTask(4, 5));
    Thread t4 = new Thread(new SumTask(6, 7));
    // 启动线程
    t1.start();
    t2.start();
    t3.start();
    t4.start();
    // 等待线程结束
    t1.join();
    t2.join();
    t3.join();
    t4.join();
    // 打印结果
    System.out.println("Sum: " + sum);
  }

  // 定义一个求和任务类，实现Runnable接口
  static class SumTask implements Runnable {
    // 定义起始和结束的索引
    int start;
    int end;

    public SumTask(int start, int end) {
      this.start = start;
      this.end = end;
    }

    @Override
    public void run() {
      // 计算部分和
      int partSum = 0;
      for (int i = start; i <= end; i++) {
        partSum += array[i];
      }
      // 同步更新结果变量
      synchronized (lock) {
        sum += partSum;
      }
    }
  }
}

// Go
package main

import (
	"fmt"
	"sync"
)

func main() {
	// 定义一个数组
	array := []int{1, 2, 3, 4, 5, 6, 7, 8}
	// 定义一个结果变量
	sum := 0
	// 定义一个互斥锁
	var mutex sync.Mutex
	// 定义一个等待组
	var wg sync.WaitGroup
	// 创建四个协程
	wg.Add(4)
	go sumTask(array[0:2], &sum, &mutex, &wg)
	go sumTask(array[2:4], &sum, &mutex, &wg)
	go sumTask(array[4:6], &sum, &mutex, &wg)
	go sumTask(array[6:8], &sum, &mutex, &wg)
	// 等待协程结束
	wg.Wait()
	// 打印结果
	fmt.Println("Sum:", sum)
}

// 定义一个求和任务函数，接收一个切片，一个结果指针，一个互斥锁指针，和一个等待组指针
func sumTask(slice []int, sum *int, mutex *sync.Mutex, wg *sync.WaitGroup) {
	// 计算部分和
	partSum := 0
	for _, v := range slice {
		partSum += v
	}
	// 同步更新结果变量
	mutex.Lock()
	*sum += partSum
	mutex.Unlock()
	// 标记任务完成
	wg.Done()
}
