## 回溯算法

构造解空间树进行处理

```java
class Solution {
    public static void main(String[] args) {
        Solution solution = new Solution();
        int[] results = solution.twoSum(
                new int[]{3, 2, 4}, 6);
        System.out.println(Arrays.toString(results));

        results = solution.twoSum(
                new int[]{2,7,11,15}, 9);
        System.out.println(Arrays.toString(results));


        results = solution.twoSum(
                new int[]{0,4,3,0}, 0);
        System.out.println(Arrays.toString(results));
    }
    
   public int[] twoSum(int[] nums, int target) {
        int[] results = new int[2];
        backtrack(nums,target,0,results,new ArrayList<>());
        return results;
    }

    private void backtrack(int[] nums, int target,int begin, int[] result,List<Integer> container){
        // 终止条件，超过了两个元素我们也不再继续找了，进行减枝
        if(container.size() == 2){
            if(target == 0){  // 收集结果
                result[0] = container.get(0);
                result[1] = container.get(1);
            }
            return;
        }

        for (int i = begin; i < nums.length; i++) {
            // 收集
            container.add(i);
            backtrack(nums,target - nums[i],i + 1,result,container);
            container.remove(container.size()-1);
        }
    }
}
/**
 * [1, 2]
 * [0, 1]
 * [0, 3]
 */
```

### 优化

处理找到一个结果之后就不在回溯

```java
public int[] twoSum(int[] nums, int target) {
        int[] results = new int[2];
        // 为了终止条件做判断
        results[0] = -1;results[1] = -1;
        backtrack(nums,target,0,results,new ArrayList<>());
        return results;
    }

    private void backtrack(int[] nums, int target,int begin, int[] result,List<Integer> container){
        // 代表找到一个就不再找了
        if(result[0] != -1 && result[1] != -1) return;
        // 终止条件，超过了两个元素我们也不再继续找了，进行减枝
        if(container.size() == 2){
            if(target == 0){  // 收集结果
                result[0] = container.get(0);
                result[1] = container.get(1);
            }
            return;
        }

        for (int i = begin; i < nums.length; i++) {
            // 收集
            container.add(i);
            backtrack(nums,target - nums[i],i + 1,result,container);
            container.remove(container.size()-1);
        }
    }
```



## 利用字典数据结构进行处理

> 字典的好处是能够快速找到值

```java
public class Solution2 {
    public static void main(String[] args) {
        Solution2 solution = new Solution2();
        int[] results = solution.twoSum(
                new int[]{3, 2, 4}, 6);
        System.out.println(Arrays.toString(results));

    }
    public int[] twoSum(int[] nums, int target) {
        int[] results = new int[2];
        Map<Integer,Integer> dict = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int n = nums[i];
            if(dict.get(target - n) != null){
                // 找到结果
                results[0] = dict.get(target - n);
                results[1] = i;
                break;
            }
            // 放入字典
            dict.put(n,i);
        }
        return results;
    }
}
/**
 * [1, 2]
 * [0, 1]
 * [0, 3]
 */
```
