# 1.2.2 箭头函数this为外层this

  1. 对于有异步代码要处理this，箭头函数非常方便
let p = {
    name: "Q10Viking",
    getName: function () {
        // 箭头函数的this为外层的this,即p实例
        // 异步代码
        setTimeout(() => {
            console.log(this.name);
        })
    }
}

p.getName();
