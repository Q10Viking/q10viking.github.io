# 1 let声明变量

# 1.1 let会形成独立的作用域

// var没有独立的作用域，异步执行时值已经是3，所以会输出3，3，3
for (var i = 0; i < 3; i++) {
    setTimeout(() => console.log(i))
}
// let 会形成独立的作用域，会输出0，1，2
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i))
}

# 2 解构赋值

# 结构上要对应⭐

# 2.1 数组解构

  1. 数组结构需要[]将变量包裹

# 2.2.1 正常解构

let arr = [1, 2, 3]
let [a, b, c] = arr
console.log(a, b, c); // 1 2 3

# 1 默认解构在函数形参中应用

    console.log(method, data);
