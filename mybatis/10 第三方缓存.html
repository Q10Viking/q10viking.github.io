    <version>1.0.0-beta2</version>

测试

@Test
    public void test(){
        try(SqlSession sqlSession = sqlSessionFactory.openSession()){
            DeptMapper mapper = sqlSession.getMapper(DeptMapper.class);
            Dept dept1 = new Dept();
            dept1.setDeptId(1);
            mapper.selectDept(dept1);

            // 存储二级缓存
            sqlSession.commit();

            Dept dept2 = new Dept();
            dept2.setDeptId(1);
            mapper.selectDept(dept2); // 会走二级缓存
        }
    }
