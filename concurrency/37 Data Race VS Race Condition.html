# Fixing race condition using lock

//  share variable
Map<String,String> loanedBooks = new HashMap<>();
// lock
ReentrantLock lock = new ReentrantLock();

// thread-1 
lock.lock();
if(!loanedBooks.containskey("book1")){
    loanedBooks.put("book1","user3")
}
lock.unlock();

// thread-2
lock.lock();
if(!loanedBooks.containskey("book1")){
    loanedBooks.put("book1","user7")
}
lock.unlock();
