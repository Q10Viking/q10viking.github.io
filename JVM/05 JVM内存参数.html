# 1. JVMеҶ…еӯҳеҸӮж•°и®ҫзҪ®

java -Xms2048M -Xmx2048M -Xmn1024M -Xss512K -XX:MetaspaceSize=256M -XX:MaxMetaspaceSize=256M -jar microservice-eureka-server.jar
1

# 1.1 е Ҷзӣёе…іJVMеҸӮж•°

еҸӮж•°иҜҙжҳҺ
-XmsеҲқе§Ӣе ҶеӨ§е°ҸпјҢй»ҳи®Өзү©зҗҶеҶ…еӯҳзҡ„1/64
-XmxжңҖеӨ§е ҶеӨ§е°ҸпјҢй»ҳи®Өзү©зҗҶеҶ…еӯҳзҡ„1/4
-Xmnж–°з”ҹд»ЈеӨ§е°Ҹ
-XX:NewRatioй»ҳи®Ө2иЎЁзӨәж–°з”ҹд»ЈеҚ е№ҙиҖҒд»Јзҡ„1/2пјҢеҚ ж•ҙдёӘе ҶеҶ…еӯҳзҡ„1/3
-XX:NewSizeи®ҫзҪ®ж–°з”ҹд»ЈеҲқе§ӢеӨ§е°Ҹ
-XX:SurvivorRatioй»ҳи®Ө8иЎЁзӨәдёҖдёӘsurvivorеҢәеҚ з”Ё1/8зҡ„EdenеҶ…еӯҳпјҢеҚі1/10зҡ„ж–°з”ҹд»ЈеҶ…еӯҳ

# е ҶеҶ…еӯҳxmxи®ҫзҪ®жіЁж„ҸдәӢйЎ№

  1. еһғеңҫеӣһж”¶зҡ„ејҖй”Җпјҡе ҶеҶ…еӯҳи¶ҠеӨ§пјҢеһғеңҫеӣһж”¶зҡ„ејҖй”ҖйҖҡеёёд№ҹдјҡеўһеҠ гҖӮеӨ§е ҶеҶ…еӯҳеҸҜиғҪйңҖиҰҒжӣҙй•ҝзҡ„еһғеңҫеӣһж”¶жҡӮеҒңж—¶й—ҙгҖӮеӣ жӯӨпјҢиҰҒжқғиЎЎеҶ…еӯҳеӨ§е°Ҹе’Ңеһғеңҫеӣһж”¶ејҖй”ҖгҖӮ
  2. JVMзүҲжң¬е’Ңеһғеңҫеӣһж”¶еҷЁпјҡдёҚеҗҢзҡ„JVMзүҲжң¬е’Ңеһғеңҫеӣһж”¶еҷЁеҸҜиғҪеҜ№еҶ…еӯҳйңҖжұӮжңүдёҚеҗҢзҡ„еҪұе“ҚгҖӮжҹҗдәӣеһғеңҫеӣһж”¶еҷЁеҸҜиғҪжӣҙйҖӮеҗҲеӨ§е ҶеҶ…еӯҳпјҢиҖҢжҹҗдәӣйҖӮз”ЁдәҺе°Ҹе ҶеҶ…еӯҳгҖӮ
  3. еә”з”ЁзЁӢеәҸзҡ„еҶ…еӯҳйңҖжұӮпјҡйҰ–е…ҲиҰҒдәҶи§Јеә”з”ЁзЁӢеәҸзҡ„еҶ…еӯҳйңҖжұӮгҖӮиҝҷеҢ…жӢ¬еә”з”ЁзЁӢеәҸзҡ„ж•°жҚ®йҮҸгҖҒе№¶еҸ‘з”ЁжҲ·ж•°гҖҒеҜ№иұЎеҲӣе»әйў‘зҺҮзӯүгҖӮдёҚеҗҢзҡ„еә”з”ЁзЁӢеәҸеҸҜиғҪйңҖиҰҒдёҚеҗҢеӨ§е°Ҹзҡ„е ҶеҶ…еӯҳгҖӮ
  4. зӣ‘жҺ§е’Ңи°ғж•ҙпјҡзӣ‘жҺ§еә”з”ЁзЁӢеәҸзҡ„еҶ…еӯҳдҪҝз”Ёжғ…еҶөпјҢдҪҝз”Ёе·Ҙе…·еҰӮJVisualVMгҖҒJConsoleзӯүжқҘи§ӮеҜҹе ҶеҶ…еӯҳзҡ„дҪҝз”Ёжғ…еҶөгҖӮж №жҚ®зӣ‘жҺ§ж•°жҚ®иҝӣиЎҢеҠЁжҖҒи°ғж•ҙ-XmxеҸӮж•°гҖӮ
  5. е№¶еҸ‘жҖ§йңҖжұӮпјҡеӨҡзәҝзЁӢеә”з”ЁзЁӢеәҸйҖҡеёёйңҖиҰҒжӣҙеӨҡзҡ„е ҶеҶ…еӯҳпјҢеӣ дёәжҜҸдёӘзәҝзЁӢйғҪйңҖиҰҒдёҖе®ҡзҡ„еҶ…еӯҳз©әй—ҙжқҘеӯҳеӮЁж Ҳеё§е’ҢеұҖйғЁеҸҳйҮҸгҖӮ

# 1.2 ж ҲJVMеҸӮж•°

еҸӮж•°иҜҙжҳҺ
-XssжҜҸдёӘзәҝзЁӢзҡ„ж ҲеӨ§е°ҸпјҢй»ҳи®ӨжҳҜ1M

# 1.3 е…ғз©әй—ҙзҡ„JVMеҸӮж•°

еҸӮж•°иҜҙжҳҺ
-XXпјҡMaxMetaspaceSizeи®ҫзҪ®е…ғз©әй—ҙжңҖеӨ§еҖјпјҢ й»ҳи®ӨжҳҜ-1пјҢ еҚідёҚйҷҗеҲ¶пјҢ жҲ–иҖ…иҜҙеҸӘеҸ—йҷҗдәҺжң¬ең°еҶ…еӯҳеӨ§е°Ҹ
-XXпјҡMetaspaceSizeжҢҮе®ҡе…ғз©әй—ҙи§ҰеҸ‘Fullgcзҡ„еҲқе§ӢйҳҲеҖј(е…ғз©әй—ҙж— еӣәе®ҡеҲқе§ӢеӨ§е°Ҹ)пјҢ д»Ҙеӯ—иҠӮдёәеҚ•дҪҚпјҢй»ҳи®ӨжҳҜ21Mе·ҰеҸіпјҢиҫҫеҲ°иҜҘеҖје°ұдјҡи§ҰеҸ‘full gcиҝӣиЎҢзұ»еһӢеҚёиҪҪпјҢ еҗҢж—¶ж”¶йӣҶеҷЁдјҡеҜ№иҜҘеҖјиҝӣиЎҢи°ғж•ҙпјҡ еҰӮжһңйҮҠж”ҫдәҶеӨ§йҮҸзҡ„з©әй—ҙпјҢ е°ұйҖӮеҪ“йҷҚдҪҺиҜҘеҖјпјӣ еҰӮжһңйҮҠж”ҫдәҶеҫҲе°‘зҡ„з©әй—ҙпјҢ йӮЈд№ҲеңЁдёҚи¶…иҝҮ-XXпјҡMaxMetaspaceSizeпјҲеҰӮжһңи®ҫзҪ®дәҶзҡ„иҜқпјү зҡ„жғ…еҶөдёӢпјҢ йҖӮеҪ“жҸҗй«ҳиҜҘеҖј

з”ұдәҺи°ғж•ҙе…ғз©әй—ҙзҡ„еӨ§е°ҸйңҖиҰҒFull GCпјҢиҝҷжҳҜйқһеёёжҳӮиҙөзҡ„ж“ҚдҪңпјҢеҰӮжһңеә”з”ЁеңЁеҗҜеҠЁзҡ„ж—¶еҖҷпјҲйЎ№зӣ®йқһеёёеӨ§пјүеҸ‘з”ҹеӨ§йҮҸFull GCпјҢйҖҡеёёйғҪжҳҜз”ұдәҺж°ёд№…д»ЈжҲ–**пјҲж–№жі•еҢәпјүе…ғз©әй—ҙ**еҸ‘з”ҹдәҶеӨ§е°Ҹи°ғж•ҙпјҢеҹәдәҺиҝҷз§Қжғ…еҶөпјҢдёҖиҲ¬е»әи®®еңЁJVMеҸӮж•°дёӯе°ҶMetaspaceSizeе’ҢMaxMetaspaceSizeи®ҫзҪ®жҲҗдёҖж ·зҡ„еҖјпјҢе№¶и®ҫзҪ®еҫ—жҜ”еҲқе§ӢеҖјиҰҒеӨ§пјҢеҜ№дәҺ8Gзү©зҗҶеҶ…еӯҳзҡ„жңәеҷЁжқҘиҜҙпјҢдёҖиҲ¬жҲ‘дјҡе°ҶиҝҷдёӨдёӘеҖјйғҪи®ҫзҪ®дёә256M

# seataзҡ„й…ҚзҪ®вқӨпёҸ

JAVA_OPT="${JAVA_OPT} -server -Xmx${JVM_XMX:="1024m"} -Xms${JVM_XMS:="1024m"} -Xmn${JVM_XMN:="512m"} -Xss${JVM_XSS:="512k"} -XX:SurvivorRatio=10 -XX:MetaspaceSize=${JVM_MetaspaceSize:="128m"} -XX:MaxMetaspaceSize=${JVM_MaxMetaspaceSize:="256m"} -XX:MaxDirectMemorySize=${JVM_MaxDirectMemorySize:=1024m} -XX:-OmitStackTraceInFastThrow -XX:-UseAdaptiveSizePolicy"
JAVA_OPT="${JAVA_OPT} -XX:+HeapDumpOnOutOfMemoryError -XX:HeapDumpPath=${BASEDIR}/logs/java_heapdump.hprof -XX:+DisableExplicitGC -XX:+CMSParallelRemarkEnabled -XX:+UseCMSInitiatingOccupancyOnly -XX:CMSInitiatingOccupancyFraction=75"

# 2. StackOverflowжј”зӨә

-Xssи®ҫзҪ®и¶Ҡе°ҸcountеҖји¶Ҡе°ҸпјҢиҜҙжҳҺдёҖдёӘзәҝзЁӢж ҲйҮҢиғҪеҲҶй…Қзҡ„ж Ҳеё§е°ұи¶Ҡе°‘пјҢдҪҶжҳҜеҜ№JVMж•ҙдҪ“жқҘиҜҙиғҪејҖеҗҜзҡ„зәҝзЁӢж•°дјҡжӣҙеӨҡ

// JVMи®ҫзҪ® -Xss128k(й»ҳи®Ө1M)
public class StackOverflowTest {

    static int count = 0;

    static void redo() {
        count++;
        redo();
    }

    public static void main(String[] args) {
        try {
            redo();
        } catch (Throwable t) {
            t.printStackTrace();
            System.out.println(count);
        }
    }
}

# JVMеҗҜеҠЁеҸӮж•°

JVMпјҲJavaиҷҡжӢҹжңәпјүзҡ„еҗҜеҠЁеҸӮж•°з”ЁдәҺй…ҚзҪ®е’Ңи°ғж•ҙJavaеә”з”ЁзЁӢеәҸзҡ„иҝҗиЎҢж—¶иЎҢдёәгҖӮд»ҘдёӢжҳҜдёҖдәӣеёёз”Ёзҡ„JVMеҗҜеҠЁеҸӮж•°пјҡ

  1. -XmxпјҡжҢҮе®ҡJavaе ҶеҶ…еӯҳзҡ„жңҖеӨ§йҷҗеҲ¶гҖӮдҫӢеҰӮпјҢ-Xmx512m иЎЁзӨәжңҖеӨ§е ҶеҶ…еӯҳдёә512е…Ҷеӯ—иҠӮгҖӮ
  2. -XmsпјҡжҢҮе®ҡJavaе ҶеҶ…еӯҳзҡ„еҲқе§ӢеӨ§е°ҸгҖӮдҫӢеҰӮпјҢ-Xms256m иЎЁзӨәеҲқе§Ӣе ҶеҶ…еӯҳдёә256е…Ҷеӯ—иҠӮгҖӮ
  3. -XssпјҡжҢҮе®ҡжҜҸдёӘзәҝзЁӢзҡ„е Ҷж ҲеӨ§е°ҸгҖӮдҫӢеҰӮпјҢ-Xss256k иЎЁзӨәжҜҸдёӘзәҝзЁӢзҡ„е Ҷж ҲеӨ§е°Ҹдёә256еҚғеӯ—иҠӮгҖӮ
  4. -XX:MaxPermSizeпјҲеҜ№дәҺJava 7еҸҠд№ӢеүҚзҡ„зүҲжң¬пјүжҲ– -XX:MaxMetaspaceSizeпјҲеҜ№дәҺJava 8еҸҠд»ҘеҗҺзҡ„зүҲжң¬пјүпјҡжҢҮе®ҡж°ёд№…д»ЈпјҲJava 7еҸҠд№ӢеүҚпјүжҲ–е…ғз©әй—ҙпјҲJava 8еҸҠд»ҘеҗҺпјүзҡ„жңҖеӨ§еӨ§е°ҸгҖӮ
  5. -XX:PermSizeпјҲеҜ№дәҺJava 7еҸҠд№ӢеүҚзҡ„зүҲжң¬пјүжҲ– -XX:MetaspaceSizeпјҲеҜ№дәҺJava 8еҸҠд»ҘеҗҺзҡ„зүҲжң¬пјүпјҡжҢҮе®ҡж°ёд№…д»ЈпјҲJava 7еҸҠд№ӢеүҚпјүжҲ–е…ғз©әй—ҙпјҲJava 8еҸҠд»ҘеҗҺпјүзҡ„еҲқе§ӢеӨ§е°ҸгҖӮ
  6. -XmnпјҡжҢҮе®ҡе№ҙиҪ»д»Јзҡ„еӨ§е°ҸгҖӮдҫӢеҰӮпјҢ-Xmn256m иЎЁзӨәе№ҙиҪ»д»ЈеӨ§е°Ҹдёә256е…Ҷеӯ—иҠӮгҖӮ
  7. -XX:SurvivorRatioпјҡжҢҮе®ҡе№ҙиҪ»д»ЈдёӯEdenеҢәдёҺSurvivorеҢәзҡ„еӨ§е°ҸжҜ”дҫӢгҖӮдҫӢеҰӮпјҢ-XX:SurvivorRatio=8 иЎЁзӨәEdenеҢәдёҺжҜҸдёӘSurvivorеҢәзҡ„еӨ§е°ҸжҜ”дҫӢдёә8:1гҖӮ
  8. -XX:NewRatioпјҡжҢҮе®ҡе№ҙиҪ»д»ЈдёҺиҖҒе№ҙд»Јзҡ„еӨ§е°ҸжҜ”дҫӢгҖӮдҫӢеҰӮпјҢ-XX:NewRatio=2 иЎЁзӨәе№ҙиҪ»д»Је’ҢиҖҒе№ҙд»Јзҡ„жҜ”дҫӢдёә1:2гҖӮ
  9. -XX:MaxGCPauseMillisпјҡи®ҫзҪ®еһғеңҫеӣһж”¶зҡ„жңҖеӨ§жҡӮеҒңж—¶й—ҙзӣ®ж ҮгҖӮдҫӢеҰӮпјҢ-XX:MaxGCPauseMillis=100 иЎЁзӨәеһғеңҫеӣһж”¶зҡ„жңҖеӨ§жҡӮеҒңж—¶й—ҙзӣ®ж Үдёә100жҜ«з§’гҖӮ
  10. -XX:ParallelGCThreadsпјҡжҢҮе®ҡе№¶иЎҢеһғеңҫеӣһж”¶зәҝзЁӢзҡ„ж•°йҮҸгҖӮдҫӢеҰӮпјҢ-XX:ParallelGCThreads=4 иЎЁзӨәдҪҝз”Ё4дёӘзәҝзЁӢиҝӣиЎҢе№¶иЎҢеһғеңҫеӣһж”¶гҖӮ
  11. -XX:+UseConcMarkSweepGCпјҡеҗҜз”Ёе№¶еҸ‘ж Үи®°жё…йҷӨеһғеңҫеӣһж”¶еҷЁгҖӮ
  12. -XX:+UseG1GCпјҡеҗҜз”ЁG1пјҲGarbage Firstпјүеһғеңҫеӣһж”¶еҷЁгҖӮ
  13. -Dproperty=valueпјҡи®ҫзҪ®Javaзі»з»ҹеұһжҖ§пјҢеҸҜд»ҘеңЁеә”з”ЁзЁӢеәҸдёӯдҪҝз”Ё System.getProperty("property") жқҘиҺ·еҸ–иҝҷдәӣеұһжҖ§зҡ„еҖјгҖӮ
